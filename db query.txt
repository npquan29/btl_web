4. comment

CREATE TABLE `comment` (
  `commentID` varchar(45) NOT NULL,
  `itemID_comment` varchar(45) DEFAULT NULL,
  `userID_comment` varchar(45) DEFAULT NULL,
  `content` varchar(15000) DEFAULT NULL,
  `createDate` date DEFAULT NULL,
  PRIMARY KEY (`commentID`),
  KEY `itemID_comment_idx` (`itemID_comment`),
  KEY `userID_comment_idx` (`userID_comment`),
  CONSTRAINT `itemID_comment` FOREIGN KEY (`itemID_comment`) REFERENCES `item` (`itemID`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `userID_comment` FOREIGN KEY (`userID_comment`) REFERENCES `user` (`userID`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
